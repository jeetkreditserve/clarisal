export interface CountryOption {
  code: string
  name: string
  dialCode: string
  defaultCurrency: string
}
 
export interface CurrencyOption {
  code: string
  label: string
}
 
const displayNamesIntl = Intl as typeof Intl & {
  DisplayNames?: new (
    locales?: string | string[],
    options?: { type: 'currency' | 'region' }
  ) => { of(code: string): string | undefined }
}
 
const currencyDisplayNames = displayNamesIntl.DisplayNames
  ? new displayNamesIntl.DisplayNames(['en'], { type: 'currency' })
  : null
 
export const COUNTRY_OPTIONS: CountryOption[] = COUNTRY_METADATA_ROWS.split('\n')
  .map((row) => {
    const [code, name, dialCode, defaultCurrency] = row.split('|')
    return { code, name, dialCode, defaultCurrency }
  })
  .sort((left, right) => left.name.localeCompare(right.name))
 
export const COUNTRY_OPTIONS_BY_CODE = COUNTRY_OPTIONS.reduce<Record<string, CountryOption>>((accumulator, country) => {
  accumulator[country.code] = country
  return accumulator
}, {})
 
export const DEFAULT_COUNTRY_CODE = 'IN'
export const DEFAULT_CURRENCY_CODE = 'INR'
export const DEFAULT_COUNTRY_OPTION = COUNTRY_OPTIONS_BY_CODE[DEFAULT_COUNTRY_CODE]
 
const uniqueCurrencyCodes = Array.from(new Set(COUNTRY_OPTIONS.map((country) => country.defaultCurrency))).sort((left, right) =>
  left.localeCompare(right),
)
 
export const CURRENCY_OPTIONS: CurrencyOption[] = uniqueCurrencyCodes.map((code) => ({
  code,
  label: formatCurrencyOptionLabel(code),
}))

export function getCountryOption(code?: string | null) {
  if (!code) return null
  return COUNTRY_OPTIONS_BY_CODE[code.toUpperCase()] ?? null
}
 
export function getCountryName(code?: string | null) {
  return getCountryOption(code)?.name ?? ''
}
 
export function getDefaultCurrencyForCountry(code?: string | null) {
  return getCountryOption(code)?.defaultCurrency ?? DEFAULT_CURRENCY_CODE
}
 
export function formatCurrencyOptionLabel(code: string) {
  const label = currencyDisplayNames?.of(code)
  return label ? `${code} - ${label}` : code
}
 
export function normalizePhoneForValidation(value: string) {
  const rawValue = value.trim()
  if (!rawValue) return ''
  return rawValue.replace(/[\s()-]+/g, '')
}
 
export function validatePhoneForCountry(phone: string, countryCode: string) {
  const normalizedPhone = normalizePhoneForValidation(phone)
  if (!normalizedPhone) return null
  if (!normalizedPhone.startsWith('+')) {
    return 'Phone number must include the country dial code prefix, for example +91.'
  }
  if (!/^\+\d+$/.test(normalizedPhone)) {
    return 'Phone number must contain only digits after the + prefix.'
  }
  const country = getCountryOption(countryCode)
  if (!country) {
    return 'Select a valid country.'
  }
  if (!normalizedPhone.startsWith(country.dialCode)) {
    return `Phone number must start with ${country.dialCode} for ${country.name}.`
  }
  return null
}
